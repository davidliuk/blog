List<List<Integer>> combine(int[] nums) {
    int n = nums.length;
    Arrays.sort(nums);
    var combs = new ArrayList<List<Integer>>();
    for (int i = 0; i < (1 << n); i++) {
        var comb = new ArrayList<Integer>();
        for (int j = 0; j < n; j++) {
            if ((i & (1 << j)) != 0) {
                comb.add(nums[j]);
            }
        }
        combs.add(comb);
    }
    return combs;
}

    int n = nums.length;
    Arrays.sort(nums);
    var perms = new ArrayList<List<Integer>>();
    while (nums != null) {
        var perm = new ArrayList<Integer>();
        for (int i = 0; i < n; i++) {
            perm.add(nums[i]);
        }
