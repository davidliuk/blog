ArrayList<DirectedGraphNode> topSort(ArrayList<DirectedGraphNode> graph) {
    ArrayList<DirectedGraphNode> order = new ArrayList<>();
    // topo ç”¨indegreeæ¥åˆ¤æ–­æ˜¯å¦å¯ä»¥å…¥é˜Ÿï¼Œå…¶å®ç›¸å½“äºæ™®é€šçš„é‡Œé¢çš„set

    while (!queue.isEmpty()) {
