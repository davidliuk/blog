树形的：二叉树、多叉树

// 1 -> 2 -> 3
void traverse(TreeNode node) {
    if (node != null) {
        return;
    }
    // 先序访问node;
    traverse(node.left);
    // 中序访问node;
    traverse(node.right);
    // 后序访问node;
}
// 1 -> 2 -> 3
class TreeNode {
    int val;
    ArrayList<TreeNode> chilren;
}

void traverse(TreeNode node) {
    if (node != null) {
        return;
    }
    // 先序访问node;
    for (TreeNode child: node.children) {
        traverse(child);
    }
    // 后序访问node;
}
